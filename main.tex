%commands for table sections
\usepackage{booktabs}
\newcommand*{\header}[2]{
    \toprule
    \rowcolor{#1} #2 \\
}
\newcommand*{\tablemid}{
    \midrule
}
\newcommand*{\tablebottom}{
    \bottomrule
}

%###########

% table enviroment
\usepackage{tabularx}
\extrarowheight=\aboverulesep
\addtolength{\extrarowheight}{\belowrulesep}
\aboverulesep=0pt
\belowrulesep=0pt
\newenvironment{specialtabular}[4]{
    \rowcolors{2}{#3}{#4}
    \tabularx{#1}{#2}
}
{
    \endtabularx
}
